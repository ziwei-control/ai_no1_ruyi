📝 使用方法

1. 引入文件

<!-- CSS 样式 -->
<link rel="stylesheet" href="themes.css">

<!-- JavaScript 脚本 -->
<script src="themes.js"></script>

2. HTML 结构

<html lang="zh-CN" data-theme="warm">
<!-- 页面内容 -->
</html>

3. 主题切换按钮会自动插入到导航栏

// 切换主题
themeManager.switchTheme('scifi'); // 切换到科幻主题

// 获取当前主题
const currentTheme = themeManager.getTheme(); // 'warm', 'scifi', 'feminine', 'cyberpunk'
